495  Geometry2d::ShapeSet result;
496  for (size_t i = 0; i < mask.size(); ++i)
497  if (mask[i] > 0 && robots[i] && robots[i]->visible) {

499  result.add(std::shared_ptr<Geometry2d::Shape>(
500  new Geometry2d::Circle(robots[i]->pos, mask[i])));
