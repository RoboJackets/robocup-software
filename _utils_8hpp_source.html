136  FIRFilter(const T& zero, size_t nrTaps) : _zero(zero) {
137  if (nrTaps == 0)
138  throw std::invalid_argument(
139  "FIR Filter: number of taps must be greater than zero");
140 
141  // initialize
142  _taps.assign(nrTaps, _zero);

144  _coeffs[0] = 1.0;
145  }

147  T filter(const T& x) {
148  _taps.push_front(x);
149  _taps.pop_back();
150 
151  T y = _zero;
152  for (size_t i = 0; i < _taps.size(); ++i)
153  y += _taps.at(i) * _coeffs.at(i);
154 
155  return y;
156  }

159  void setCoeffs(const Coeffs& coeffs) {
160  size_t nrTaps = coeffs.size();
161  if (nrTaps == 0)
162  throw std::invalid_argument(
163  "FIR Filter: number of coeffs must be greater than zero");
164 
165  _taps.assign(nrTaps, _zero);
166  _coeffs.assign(nrTaps, 0.0);
167 
168  // find the normalizer
169  float normalizer = 0.0;
170  for (size_t i = 0; i < coeffs.size(); ++i) normalizer += coeffs.at(i);
171 
172  // set the normalized coefficients
173  for (size_t i = 0; i < coeffs.size(); ++i)
174  _coeffs[i] = coeffs.at(i) / normalizer;
175  }
